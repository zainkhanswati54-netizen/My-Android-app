    "# ── CELL 3: Download voice samples ────────────────────────\n",
    "import urllib.request, os, struct, math\n",
    "\n",
    "os.makedirs('/tmp/ref', exist_ok=True)\n",
    "\n",
    "MALE_URLS = [\n",
    "    'https://huggingface.co/spaces/coqui/xtts/resolve/main/examples/male.wav',\n",
    "    'https://huggingface.co/coqui/XTTS-v2/resolve/main/samples/en_sample.wav',\n",
    "]\n",
    "FEMALE_URLS = [\n",
    "    'https://huggingface.co/spaces/coqui/xtts/resolve/main/examples/female.wav',\n",
    "]\n",
    "\n",
    "def download(urls, path):\n",
    "    for url in urls:\n",
    "        try:\n",
    "            req = urllib.request.Request(url, headers={'User-Agent': 'Mozilla/5.0'})\n",
    "            with urllib.request.urlopen(req, timeout=20) as r:\n",
    "                data = r.read()\n",
    "            if len(data) > 5000:\n",
    "                with open(path, 'wb') as f:\n",
    "                    f.write(data)\n",
    "                print(f'✅ Downloaded: {path} ({len(data)} bytes)')\n",
    "                return True\n",
    "        except Exception as e:\n",
    "            print(f'⚠️ {url}: {e}')\n",
    "    return False\n",
    "\n",
    "def make_tone(path, freq=110.0, duration=5.0, rate=22050):\n",
    "    n = int(rate * duration)\n",
    "    samples = [int(32767 * math.sin(2*math.pi*freq*i/rate)) for i in range(n)]\n",
    "    with open(path, 'wb') as f:\n",
    "        f.write(struct.pack('<4sI4s', b'RIFF', 36+n*2, b'WAVE'))\n",
    "        f.write(struct.pack('<4sIHHIIHH', b'fmt ', 16, 1, 1, rate, rate*2, 2, 16))\n",
    "        f.write(struct.pack('<4sI', b'data', n*2))\n",
    "        for s in samples:\n",
    "            f.write(struct.pack('<h', s))\n",
    "\n",
    "if not download(MALE_URLS, '/tmp/ref/male.wav'):\n",
    "    make_tone('/tmp/ref/male.wav', 110.0)\n",
    "if not download(FEMALE_URLS, '/tmp/ref/female.wav'):\n",
    "    make_tone('/tmp/ref/female.wav', 220.0)\n",
    "\n",
    "print('✅ Voice samples ready!')"

    "# ── CELL 5: FastAPI Server ─────────────────────────────────\n",
    "import tempfile, traceback, re, wave, asyncio\n",
    "import nest_asyncio\n",
    "nest_asyncio.apply()\n",
    "\n",
    "from fastapi import FastAPI, Query\n",
    "from fastapi.responses import FileResponse, JSONResponse, HTMLResponse\n",
    "import uvicorn\n",
    "\n",
    "app = FastAPI()\n",
    "\n",
    "VOICE_MAP = {\n",
    "    'en-Male': ('male', 'en'), 'en-Female': ('female', 'en'),\n",
    "    'hi-Male': ('male', 'hi'), 'hi-Female': ('female', 'hi'),\n",
    "    'ur-Male': ('male', 'ur'), 'ur-Female': ('female', 'ur'),\n",
    "    'ur-PK-AsadNeural': ('male', 'ur'), 'ur-PK-UzmaNeural': ('female', 'ur'),\n",
    "    'hi-IN-MadhurNeural': ('male', 'hi'), 'hi-IN-SwaraNeural': ('female', 'hi'),\n",
    "}\n",
    "\n",
    "EMOTION_PARAMS = {\n",
    "    'Normal': {'speed': 1.0}, 'Happy': {'speed': 1.1},\n",
    "    'Sad': {'speed': 0.85}, 'Excited': {'speed': 1.2},\n",
    "    'Calm': {'speed': 0.9}, 'Serious': {'speed': 1.0},\n",
    "    'Whisper': {'speed': 0.85}, 'Shout': {'speed': 1.1},\n",
    "}\n",
    "\n",
    "AZURE_VOICES = {'male': 'ur-PK-AsadNeural', 'female': 'ur-PK-UzmaNeural'}\n",
    "\n",
    "async def azure_urdu(text, gender, output_path):\n",
    "    try:\n",
    "        import edge_tts, subprocess\n",
    "        voice = AZURE_VOICES.get(gender, AZURE_VOICES['male'])\n",
    "        mp3_tmp = output_path.replace('.wav', '_az.mp3')\n",
    "        await edge_tts.Communicate(text=text, voice=voice).save(mp3_tmp)\n",
    "        result = subprocess.run(\n",
    "            ['ffmpeg', '-y', '-i', mp3_tmp, '-ar', '22050', '-ac', '1', output_path],\n",
    "            capture_output=True, timeout=30\n",
    "        )\n",
    "        if os.path.exists(mp3_tmp): os.unlink(mp3_tmp)\n",
    "        return result.returncode == 0 and os.path.getsize(output_path) > 500\n",
    "    except Exception as e:\n",
    "        print(f'Azure error: {e}')\n",
    "        return False\n",
    "\n",
    "def split_chunks(text, max_len=180):\n",
    "    if len(text) <= max_len: return [text]\n",
    "    chunks, remaining = [], text.strip()\n",
    "    while remaining:\n",
    "        if len(remaining) <= max_len:\n",
    "            chunks.append(remaining.strip())\n",
    "            break\n",
    "        seg = remaining[:max_len]\n",
    "        split_at = -1\n",
    "        for d in ['. ', '! ', '? ', '۔ ', ', ', ' ']:\n",
    "            idx = seg.rfind(d)\n",
    "            if idx > max_len // 3:\n",
    "                split_at = idx + len(d)\n",
    "                break\n",
    "        if split_at == -1: split_at = max_len\n",
    "        chunks.append(remaining[:split_at].strip())\n",
    "        remaining = remaining[split_at:].strip()\n",
    "    return [c for c in chunks if c]\n",
    "\n",
    "def merge_wavs(paths, out):\n",
    "    data, params = [], None\n",
    "    for p in paths:\n",
    "        with wave.open(p, 'rb') as f:\n",
    "            if params is None: params = f.getparams()\n",
    "            data.append(f.readframes(f.getnframes()))\n",
    "    with wave.open(out, 'wb') as f:\n",
    "        f.setparams(params)\n",
    "        for d in data: f.writeframes(d)\n",
    "\n",
    "@app.get('/', response_class=HTMLResponse)\n",
    "def home():\n",
    "    return '<h2>🎙️ Titan TTS — Kaggle GPU Server</h2><p>Status: ✅ Running</p>'\n",
    "\n",
    "@app.get('/health')\n",
    "def health():\n",
    "    return {'status': 'ok', 'gpu': str(torch.cuda.is_available()),\n",
    "            'device': torch.cuda.get_device_name(0) if torch.cuda.is_available() else 'CPU'}\n",
    "\n",
    "@app.get('/tts')\n",
    "async def tts_endpoint(\n",
    "    text: str = Query(...),\n",
    "    voice: str = Query('en-Male'),\n",
    "    emotion: str = Query('Normal'),\n",
    "    rate: int = Query(0),\n",
    "    pitch: int = Query(0),\n",
    "    volume: int = Query(0),\n",
    "    character: str = Query(''),\n",
    "    preset: str = Query(''),\n",
    "    low_latency: bool = Query(False),\n",
    "):\n",
    "    try:\n",
    "        text = text.strip()\n",
    "        if not text:\n",
    "            return JSONResponse({'error': 'Empty text'}, status_code=400)\n",
    "\n",
    "        max_chars = 150 if low_latency else 1000\n",
    "        if len(text) > max_chars:\n",
    "            text = text[:max_chars]\n",
    "\n",
    "        mapping = VOICE_MAP.get(voice)\n",
    "        if mapping:\n",
    "            gender, lang = mapping\n",
    "        else:\n",
    "            gender = 'female' if any(x in voice.lower() for x in ['female','jenny','swara','uzma','nova','zara','luna','aria']) else 'male'\n",
    "            lang = 'ur' if 'ur' in voice.lower() else ('hi' if 'hi' in voice.lower() else 'en')\n",
    "\n",
    "        ref_path = f'/tmp/ref/{gender}.wav'\n",
    "        speed_mult = EMOTION_PARAMS.get(emotion, {'speed': 1.0})['speed']\n",
    "        speed = max(0.5, min(2.0, 1.0 + (rate/100.0)*0.8)) * speed_mult\n",
    "\n",
    "        tmp = tempfile.NamedTemporaryFile(suffix='.wav', delete=False, dir='/tmp')\n",
    "        tmp.close()\n",
    "\n",
    "        # Urdu: XTTS try → Azure fallback\n",
    "        if lang == 'ur':\n",
    "            try:\n",
    "                tts.tts_to_file(text=text, file_path=tmp.name,\n",
    "                               speaker_wav=ref_path, language='ur', speed=speed)\n",
    "                if os.path.getsize(tmp.name) < 500:\n",
    "                    raise Exception('Empty output')\n",
    "                print('✅ XTTS Urdu OK')\n",
    "            except:\n",
    "                print('⚠️ XTTS Urdu failed, Azure fallback...')\n",
    "                ok = await azure_urdu(text, gender, tmp.name)\n",
    "                if not ok:\n",
    "                    return JSONResponse({'error': 'Urdu TTS failed'}, status_code=500)\n",
    "        else:\n",
    "            chunks = split_chunks(text)\n",
    "            if len(chunks) == 1:\n",
    "                tts.tts_to_file(text=chunks[0], file_path=tmp.name,\n",
    "                               speaker_wav=ref_path, language=lang, speed=speed)\n",
    "            else:\n",
    "                chunk_paths = []\n",
    "                for chunk in chunks:\n",
    "                    ct = tempfile.NamedTemporaryFile(suffix='.wav', delete=False, dir='/tmp')\n",
    "                    ct.close()\n",
    "                    tts.tts_to_file(text=chunk, file_path=ct.name,\n",
    "                                   speaker_wav=ref_path, language=lang, speed=speed)\n",
    "                    if os.path.getsize(ct.name) > 100:\n",
    "                        chunk_paths.append(ct.name)\n",
    "                if chunk_paths:\n",
    "                    merge_wavs(chunk_paths, tmp.name)\n",
    "                    for p in chunk_paths:\n",
    "                        try: os.unlink(p)\n",
    "                        except: pass\n",
    "\n",
    "        size = os.path.getsize(tmp.name)\n",
    "        print(f'✅ Generated: {size} bytes | lang={lang} | emotion={emotion}')\n",
    "        return FileResponse(tmp.name, media_type='audio/wav', filename='tts.wav')\n",
    "\n",
    "    except Exception as e:\n",
    "        print(traceback.format_exc())\n",
    "        return JSONResponse({'error': str(e)}, status_code=500)\n",
    "\n",
    "print('✅ FastAPI app ready!')"
